// gui/app.js
class MMSSGUI {
    constructor() {
        this.currentConfig = null;
        this.loadGUIData();
    }

    async loadGUIData() {
        try {
            const response = await fetch('../gui-data.json');
            const data = await response.json();
            this.updateGUI(data);
        } catch (error) {
            console.log('GUI data not available yet');
        }
    }

    updateGUI(data) {
        // Обновляем информацию о MMSS
        document.getElementById('mmssInfo').innerHTML = `
            <strong>System:</strong> ${data.overview.system}<br>
            <strong>Version:</strong> ${data.overview.version}<br>
            <strong>Clusters:</strong> ${data.overview.clusters}
        `;

        // Обновляем кластеры
        const clustersContainer = document.getElementById('clustersContainer');
        clustersContainer.innerHTML = data.cluster_breakdown.map(cluster => `
            <div class="cluster">
                <strong>${cluster.label}</strong> (${cluster.type})<br>
                <small>${cluster.prompts.join(', ')}</small>
            </div>
        `).join('');

        // Обновляем параметры
        document.getElementById('lyricsValue').textContent = data.parameters.lyrics_strength.toFixed(3);
        document.getElementById('lyricsStrength').value = data.parameters.lyrics_strength;
        
        document.getElementById('weirdnessValue').textContent = data.parameters.weirdness.toFixed(3);
        document.getElementById('weirdness').value = data.parameters.weirdness;

        // Обновляем звуковые промпты
        document.getElementById('soundPrompts').value = data.sound_prompts.join(', ');

        // Обновляем ML инсайты
        document.getElementById('resultsContainer').innerHTML = `
            <div class="cluster">
                <h3>ML Insights</h3>
                <p>Training Data: ${data.ml_insights.training_data} samples</p>
                <p>Confidence: ${data.ml_insights.confidence}</p>
            </div>
        `;
    }

    handleMMSSUpload(event) {
        const file = event.target.files[0];
        if (file) {
            const reader = new FileReader();
            reader.onload = (e) => {
                try {
                    this.currentConfig = JSON.parse(e.target.result);
                    this.processNewConfig();
                } catch (error) {
                    alert('Error parsing MMSS config: ' + error.message);
                }
            };
            reader.readAsText(file);
        }
    }

    async processNewConfig() {
        // Здесь будет вызов к бэкенду для обработки новой конфигурации
        alert('New MMSS config loaded! System: ' + this.currentConfig.root.label);
    }
}

// Глобальные функции для HTML
function generateMusic() {
    const prompts = document.getElementById('soundPrompts').value;
    const lyricsStrength = document.getElementById('lyricsStrength').value;
    const weirdness = document.getElementById('weirdness').value;
    
    alert(`Generating music with:\nPrompts: ${prompts}\nLyrics: ${lyricsStrength}\nWeirdness: ${weirdness}`);
}

// Инициализация GUI
const gui = new MMSSGUI();

// Обработчики событий
document.getElementById('mmssUpload').addEventListener('change', (e) => gui.handleMMSSUpload(e));
document.getElementById('lyricsStrength').addEventListener('input', (e) => {
    document.getElementById('lyricsValue').textContent = e.target.value;
});
document.getElementById('weirdness').addEventListener('input', (e) => {
    document.getElementById('weirdnessValue').textContent = e.target.value;
});
