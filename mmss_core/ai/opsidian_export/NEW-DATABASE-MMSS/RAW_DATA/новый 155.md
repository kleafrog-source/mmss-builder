// test-enhanced-system.js
import { readFileSync, writeFileSync } from 'fs';
import { IntegrationEngine } from './core/integration-engine.js';
import { ClusterProfileManager } from './core/profiles/cluster-manager.js';
import { MLOptimizer } from './core/optimizers/ml-optimizer.js';
import { RealTimeAdapter } from './core/adapters/realtime-adapter.js';

console.log('🚀 TESTING ENHANCED MMSS-PRODUCER SYSTEM');
console.log('=========================================');

async function testEnhancedSystem() {
  try {
    // 1. Загружаем MMSS конфиг
    console.log('\n📖 Loading MMSS configuration...');
    const mmssConfig = JSON.parse(readFileSync('mmss-config.json', 'utf8'));
    console.log(`✅ MMSS loaded: ${mmssConfig.root.label}`);

    // 2. Тестируем Cluster Profile Manager
    console.log('\n🎵 TESTING CLUSTER PROFILE MANAGER');
    console.log('=================================');
    const clusterManager = new ClusterProfileManager();
    const clusterAnalysis = clusterManager.generateClusterPrompts(mmssConfig);
    console.log('📊 Cluster Analysis:');
    console.log(`   - Sound Prompts: ${clusterAnalysis.sound_prompts.slice(0, 3).join(', ')}...`);
    console.log(`   - Lyrics Strength: ${clusterAnalysis.parameters.lyrics_strength.toFixed(3)}`);
    console.log(`   - Weirdness: ${clusterAnalysis.parameters.weirdness.toFixed(3)}`);

    const clusterFlows = clusterManager.createClusterSpecificFlows(mmssConfig);
    console.log(`   - Cluster Flows: ${clusterFlows.length} flows generated`);

    // 3. Тестируем ML Optimizer
    console.log('\n🤖 TESTING ML OPTIMIZER');
    console.log('======================');
    const mlOptimizer = new MLOptimizer();
    const mlParams = mlOptimizer.suggestOptimalParameters(mmssConfig, 'balanced');
    console.log('📈 ML Optimization Results:');
    console.log(`   - Lyrics Strength: ${mlParams.lyrics_strength.toFixed(3)}`);
    console.log(`   - Weirdness: ${mlParams.weirdness.toFixed(3)}`);
    console.log(`   - Training Data: ${mlOptimizer.trainingData.length} samples`);

    // 4. Тестируем Real-Time Adapter
    console.log('\n⚡ TESTING REAL-TIME ADAPTER');
    console.log('===========================');
    const realtimeAdapter = new RealTimeAdapter();
    const initialParams = realtimeAdapter.initializeSession(mmssConfig, mlParams);
    
    // Симулируем real-time метрики
    const testMetrics = {
      entropy: 7.5,
      coherence: 7.8,
      energy: 0.85
    };
    
    const adaptation = realtimeAdapter.processRealtimeMetrics(testMetrics);
    console.log('🔄 Real-time Adaptation:');
    console.log(`   - Adaptations Applied: ${adaptation.adaptations.length}`);
    console.log(`   - Current Parameters:`, adaptation.adapted_parameters);

    // 5. Тестируем полную интеграцию
    console.log('\n🏗️ TESTING INTEGRATION ENGINE');
    console.log('============================');
    const integrationEngine = new IntegrationEngine();
    const integratedResult = await integrationEngine.processMMSSConfig(mmssConfig, {
      optimizationProfile: 'balanced'
    });

    console.log('🎯 Integration Results:');
    console.log(`   - Version: ${integratedResult.version}`);
    console.log(`   - Supported: ${integratedResult.metadata.supported_version}`);
    console.log(`   - Total Clusters: ${integratedResult.metadata.total_clusters}`);
    console.log(`   - Sound Prompts: ${integratedResult.sound_prompts.length}`);
    console.log(`   - Adaptation Ready: ${integratedResult.adaptation_ready}`);

    // 6. Сохраняем результаты
    console.log('\n💾 SAVING RESULTS...');
    const testResults = {
      test_timestamp: new Date().toISOString(),
      mmss_system: mmssConfig.root.label,
      cluster_analysis: clusterAnalysis,
      ml_optimization: mlParams,
      realtime_adaptation: adaptation,
      integration_result: integratedResult,
      system_components: {
        cluster_profiles: Object.keys(clusterManager.profiles).length,
        ml_training_data: mlOptimizer.trainingData.length,
        adaptation_rules: Object.keys(realtimeAdapter.adaptationRules).length
      }
    };

    writeFileSync('enhanced-system-results.json', JSON.stringify(testResults, null, 2));
    writeFileSync('optimized-generation-request.json', JSON.stringify({
      sound_prompts: integratedResult.sound_prompts,
      parameters: integratedResult.parameters,
      clusters: integratedResult.cluster_flows.length,
      version: integratedResult.version
    }, null, 2));

    console.log('✅ Enhanced system test completed!');
    console.log('📁 Generated files:');
    console.log('   - enhanced-system-results.json');
    console.log('   - optimized-generation-request.json');

    // 7. Генерируем отчет для GUI
    generateGUIReport(testResults);

  } catch (error) {
    console.log('❌ Test failed:', error.message);
    console.log('Stack:', error.stack);
  }
}

function generateGUIReport(testResults) {
  const guiReport = {
    overview: {
      system: testResults.mmss_system,
      version: testResults.integration_result.version,
      clusters: testResults.integration_result.metadata.total_clusters,
      optimization_profile: 'balanced'
    },
    parameters: {
      lyrics_strength: testResults.integration_result.parameters.lyrics_strength,
      weirdness: testResults.integration_result.parameters.weirdness,
      sound_prompt_strength: testResults.integration_result.parameters.sound_prompt_strength
    },
    sound_prompts: testResults.integration_result.sound_prompts,
    cluster_breakdown: testResults.integration_result.cluster_flows.map(flow => ({
      type: flow.cluster_type,
      label: flow.cluster_label,
      prompts: flow.sound_prompts.slice(0, 2)
    })),
    ml_insights: {
      training_data: testResults.system_components.ml_training_data,
      confidence: testResults.system_components.ml_training_data > 0 ? 'medium' : 'low'
    }
  };

  writeFileSync('gui-data.json', JSON.stringify(guiReport, null, 2));
  console.log('   - gui-data.json (for web interface)');
}

// Запуск теста
testEnhancedSystem();
