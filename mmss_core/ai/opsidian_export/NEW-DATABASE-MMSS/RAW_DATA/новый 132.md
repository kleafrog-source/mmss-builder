        "Initiate": "🌀", "Reject": "❌", "Bitterness": "😣",
        "Withdraw": "🕳️", "Invitation": "📩", "Success": "✅"
      };
      return map[phase] || phase;
    }

    function renderTooltip(text, desc) {
      return `<span class="tooltip">${text}<span class="tooltip-text">${desc}</span></span>`;
    }

    function copyToClipboard(text) {
      if (navigator.clipboard) {
        navigator.clipboard.writeText(text).catch(err => console.error('Ошибка копирования:', err));
      } else {
        const textarea = document.createElement('textarea');
        textarea.value = text;
        document.body.appendChild(textarea);
        textarea.select();
        document.execCommand('copy');
        document.body.removeChild(textarea);
      }
    }

    // === РЕНДЕР ===
    let currentDate = new Date();
    if (currentDate.getFullYear() !== 2025 || currentDate.getMonth() !== 9) {
      currentDate = new Date('2025-10-01');
    }

    function renderReport(date) {
      const dateStr = date.toISOString().slice(0, 10);
      const displayDate = date.toLocaleDateString('ru-RU', {
        weekday: 'long', day: 'numeric', month: 'long', year: 'numeric'
      });
      document.getElementById('date').innerText = displayDate;

      const report = dailyReports[dateStr];
      if (!report) {
        document.getElementById('seedQuote').innerText = "Нет данных на эту дату.";
        document.getElementById('metaBar').innerHTML = "";
        // Очистка остального — по желанию
        return;
      }

      // Семя дня
      document.getElementById('seedQuote').innerText = `"${report.core_tip}"`;

      // Мета-панель
      const phaseEmoji = getChaosPhaseEmoji(report.chaos_phase);
      const resonanceClass = getResonanceClass(report.resonance_score);
      document.getElementById('metaBar').innerHTML = `
        <div class="meta-tag">${phaseEmoji} ${report.chaos_phase}</div>
        <div class="meta-tag">${report.symbol}</div>
        <div class="meta-tag ${resonanceClass}">💎 ${report.resonance_score}</div>
      `;

      // Основные блоки
      document.getElementById('strategy').innerText = report.strategy;
      document.getElementById('do').innerText = report.do;
      document.getElementById('avoid').innerText = report.avoid;

      // Измерения с тултипами
      document.getElementById('dimensions').innerHTML = report.dimensions
        .map(dim => {
          const desc = dimensionFormulas[dim] || "Описание недоступно";
          return `<div class="dimension-item">${renderTooltip(dim, desc)}</div>`;
        })
        .join('');

      // Модальности
      document.getElementById('modalities').innerHTML = Object.entries(report.modalities)
        .map(([key, val]) => `<div class="modality"><strong>${key}:</strong> ${val}</div>`)
        .join('');

      // Мета-формулы с тултипами
      document.getElementById('formulas').innerHTML = report.meta_formulas
        .map(mf => {
          const data = metaFormulasDB[mf] || { formula: mf, desc: "Описание недоступно" };
          const tooltipText = `${data.formula}<br>${data.desc}`;
          return renderTooltip(mf, tooltipText);
        })
        .join(', ');

      // Кнопка копирования
      document.getElementById('copyBtn').onclick = () => {
        const text = `
Мета-Отчёт на ${displayDate}
"${report.core_tip}"

Стратегия: ${report.strategy}
Поддерживать: ${report.do}
Избегать: ${report.avoid}

Резонанс: ${report.resonance_score}
Фаза хаоса: ${report.chaos_phase}
Символ: ${report.symbol}

Мета-формулы: ${report.meta_formulas.join(', ')}
        `.trim();
        copyToClipboard(text);
        const feedback = document.getElementById('copyFeedback');
        feedback.innerText = "Скопировано!";
        setTimeout(() => feedback.innerText = "", 2000);
      };
    }

    // Навигация
    document.getElementById('prevDay').onclick = () => {
      const newDate = new Date(currentDate);
      newDate.setDate(currentDate.getDate() - 1);
      if (newDate >= new Date('2025-10-01')) {
        currentDate = newDate;
        renderReport(currentDate);
      }
    };

    document.getElementById('nextDay').onclick = () => {
      const newDate = new Date(currentDate);
      newDate.setDate(currentDate.getDate() + 1);
      if (newDate <= new Date('2025-10-31')) {
        currentDate = newDate;
        renderReport(currentDate);
      }
    };

    // Инициализация
    renderReport(currentDate);
  </script>
